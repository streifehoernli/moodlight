103  CMU_ClockEnable(cmuClock_TIMER0, true); // enable timer clock
104  /* load default values for general TIMER configuration */
105  TIMER_Init_TypeDef timerInit = TIMER_INIT_DEFAULT;
106  TIMER_Init(TIMER0, &timerInit); // init and start the timer
107  TIMER_TopSet(TIMER0, value_top); // TOP defines PWM period
108  /* load default values for compare/capture channel configuration */
109  TIMER_InitCC_TypeDef timerInitCC = TIMER_INITCC_DEFAULT;
110  timerInitCC.mode = timerCCModePWM; // configure as PWM channel
111  TIMER_InitCC(TIMER0, 0, &timerInitCC); // CC channel 0 is used
112  TIMER_CompareSet(TIMER0, 0, value_compare); // CC value defines PWM active time
113  /* route output to location #3 and enable output CC0 */
114  TIMER0 ->ROUTE |= (TIMER_ROUTE_LOCATION_LOC3 | TIMER_ROUTE_CC0PEN);
115  NVIC_ClearPendingIRQ(TIMER0_IRQn); // clear pending timer interrupts
116  TIMER_IntEnable(TIMER0, TIMER_IF_OF); // enable timer overflow interrupt
117  NVIC_EnableIRQ(TIMER0_IRQn); // enable timer interrupts
118 }

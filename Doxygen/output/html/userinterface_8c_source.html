188 void UI_FSM_event_RemoteControl(void) {
189  char command[COM_BUF_SIZE] = ""; // receive buffer
190  if (COM_RX_Available()) { // check for a new command string
191  COM_RX_GetData(command, COM_BUF_SIZE);
192  /* loop through all valid states to check if one matches */
193  for (uint32_t state = 0; state < UI_STATE_COUNT; state++) { // loop through states
194  /* check if the command contains a valid state */
195  if (0 == strncmp(UI_text[state], command, UI_TEXT_COMPARE_LENGTH)) {
196  UI_state_next = state; // change to that state
197  /* check if the command contains a valid number */
198  char *pos, *end; // temporary variables for char position and end
199  /* the number (if any) is separated by a ' ' from the state */
200  pos = strchr(command, ' '); // so look for a ' '
201  if (pos) { // a ' ' has been found
202  /* try to convert the remainder of the command into a number */
203  int32_t number = strtol(pos, &end, 10); // convert, base 10
204  if ('\0' == *end) { // conversion was successful
205  UI_value_next = number;
206  UI_value_changed = true; // set the value changed flag
207  }
208  }
209  }
210  }
211  UI_state_changed = true; // set the state changed flag
212  }

263  char value_string[COM_BUF_SIZE];

268  strncat(message, value_string, COM_BUF_SIZE);

270  break;

272  case START:
273  /* display state, blank display for value*/

275  SegmentLCD_NumberOff();

277  COM_TX_PutData(UI_text[UI_state_next], COM_BUF_SIZE);
278  break;
279  default:

281  }
282  }

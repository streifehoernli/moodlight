174 void LEUART0_IRQHandler(void) {
175  if (COM_LEUART->STATUS & LEUART_STATUS_RXDATAV) {// Check for RX data valid
176  char new_char = COM_LEUART->RXDATA; // Fetch the newly received char
177  if ((COM_END_OF_STRING != new_char) && (COM_BUF_SIZE > RX_index)) { // A "normal" char
178  RX_buf[RX_index] = new_char; // store the received char
179  RX_index++; // and increment the index
180  } else { // End of string is reached
181  RX_buf[RX_index] = '\0'; // Put end of string char
182  strncpy(COM_RX_Data, RX_buf, COM_BUF_SIZE); // copy new request
183  COM_RX_Available_Flag = true; // ready for processing
184  RX_index = 0; // Start a new string
185  }
186  }
187 
188  if (COM_LEUART->STATUS & LEUART_STATUS_TXBL) {// Check TX buffer level status
189  if (COM_TX_Busy_Flag) { // For synchronization
190  char next_char = TX_buf[TX_index];
191  if (('\0' != next_char) && (COM_BUF_SIZE > TX_index)) { // A "normal" char
192  COM_LEUART->TXDATA = (uint8_t) next_char; // Send the char
193  TX_index++; // and increment the index
194  } else { // End of string is reached
195  COM_LEUART->TXDATA = (uint8_t) COM_END_OF_STRING;// Send end of string char
196  COM_LEUART->IEN &= ~(LEUART_IEN_TXBL); // Disable interrupt
197  TX_index = 0; // and start anew
198  COM_TX_Busy_Flag = false;// Clear flag to avoid restart of sending
